#!/bin/sh

LOGS_PATH="/home/richmail/logs"
FIND="/usr/bin/find"
REPORT_PATH="/home/richmail/logs/report"

if [ ! -d $LOGS_PATH/backup ]; then
mkdir -p $LOGS_PATH/backup
fi

find /home/richmail/logs -maxdepth 1 -type f -name '*.log*' -mtime +3 |xargs -r mv --target-directory=/home/richmail/logs/backup

if [ $? -ne 0 ]; then
exit $?
fi

find /home/richmail/logs -maxdepth 1 -type d -name 'antispam*' -mtime +3 |xargs -r mv --target-directory=/home/richmail/logs/backup

if [ $? -ne 0 ]; then
exit $?
fi

find /home/richmail/logs/backup -type f -mtime +90 |xargs -r rm -f

if [ $? -ne 0 ]; then
exit $?
fi

find /home/richmail/logs/backup -type f ! -name '*.gz' ! -name '*.bz2' |xargs -r gzip -9

find /home/richmail/logs/backup -type d -empty -name 'antispam*' -mtime +90 |xargs -r rm -r
