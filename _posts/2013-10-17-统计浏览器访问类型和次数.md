#!/bin/bash

rm 0loginbrowser*

date=`echo 2013-10-{0{8,9},1{0..7}}`

echo "浏览器类型 $date"|tr ' ' '\n'|iconv -f utf8 -t gbk>0loginbrowser.date

echo "Opera,Chrome,Firefox,Safari,搜狗,QQ浏览器,遨游浏览器,360安全浏览器,360极速浏览器,MSIE 6.0,MSIE 7.0,MSIE 8.0,MSIE 9.0,MSIE 10.0"|iconv -f utf8 -t gbk>0loginbrowser.csv

for j in $date

do

zgrep '/webmail/login/login.do' /home/richmail/logs/mail.nginx.log|grep -v jsession|awk -F'\\||;' 'BEGIN{OFS=","}/Opera/{a++;next}/Chrome/{b++;next}/Firefox/{c++;next}/Safari/{d++;next}/MetaSr/{e++;next}/QQBrowser/{f++;next}/Maxthon/{g++;next}/360SE/{h++;next}/360EE/{i++;next}/MSIE 6.0/{j++;next}/MSIE 7.0/{k++;next}/MSIE 8.0/{l++;next}/MSIE 9.0/{m++;next}/MSIE 10.0/{n++}END{print a,b,c,d,e,f,g,h,i,j,k,l,m,n}'>>0loginbrowser.csv

done

paste -d"," 0loginbrowser.date 0loginbrowser.csv>0loginbrowsers.csv
