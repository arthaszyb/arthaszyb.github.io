#!/bin/bash

BackupPath=/data/backup/richmail/

SourcePath=/home/richmail/

Log_Sync=/data/backup/richmail/rsync_logs

Date_time=`date +%y%m%d`

if [ -d $Log_Sync ];then

rsync -avzu --delete ${SourcePath} ${BackupPath} 2>
&
1 >>${Log_Sync}/${Date_time}.log

echo "#########################################################################">>${Log_Sync}/${Date_time}.log

else echo "mkdir $Log_Sync ..."
&
&
mkdir -p $Log_Sync
&
&
echo "mkdir success!"

rsync -avzu --delete ${SourcePath} ${BackupPath} 2>
&
1 >>${Log_Sync}/${Date_time}.log
&
&
echo "########################################################################">>${Log_Sync}/${Date_time}.log

fi

#在一次全量备份后，可以不增量备份richmail/logs/，加上参数 --exclude=/home/richmail/logs
