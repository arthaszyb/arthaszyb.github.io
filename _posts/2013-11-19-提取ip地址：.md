提取ip地址：

[root@web41 ~]# ifconfig|grep "inet addr"|grep -o "[0-9.]\{1,\}"|head -n 1

192.168.70.41

[root@web41 ~]# ifconfig|grep "inet addr"|grep -o "[0-9.]\{1,\}"|sed -n 1p

192.168.70.41
